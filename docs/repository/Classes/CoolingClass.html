148    @kernel
149    def init_aoms(self, on=False):
150        """
151        Initialize AOMs by setting their frequency, scale, and attenuation. Optionally turn them on or off.
152
153        Args:
154            on (bool): If True, turn the AOMs on. If False, turn them off.
155        """
156        delay(50*ms)
157        self.urukul1_cpld.init()
158
159        for i in range(len(self.AOMs)):
160            delay(1*ms)
161
162            ch = self.urukul_channels[i]
163            ch.init()
164
165            set_f = ch.frequency_to_ftw(self.freqs[i])
166            set_asf = ch.amplitude_to_asf(self.scales[i])
167            ch.set_mu(set_f, asf=set_asf)
168            ch.set_att(self.attens[i])
169            if on:
170                ch.sw.on()
171            else:
172                ch.sw.off()
173        delay(50*ms)

222    def set_AOM_freqs(self, freq_list): # takes in a list of tuples
223        """
224        Update frequencies for specified AOMs.
225
226        Args:
227            freq_list (list): List of tuples where each tuple contains the AOM name and the new frequency.

231                ind = self.index_artiq(aom)
232                self.freqs[ind] = freq

234                set_freq = ch.frequency_to_ftw(freq)
235                set_asf = ch.amplitude_to_asf(self.scales[ind])
236                ch.set_mu(set_freq, asf=set_asf)

286        """
287        Prepare parameters for coil operation, including ramp duration and current settings.
288        """

339    @kernel
340    def Blackman_ramp_up(self, cur=-1.0):
341        if cur == -1.0: cur = self.bmot_current
342        for step in range(0, int((self.Npoints+1)//2)):
343            self.dac_set(0, cur*self.window[step])
344            delay(self.dt)
345
346    @kernel
347    def Blackman_ramp_down(self, cur=-1.0):
348        if cur == -1.0: cur = self.bmot_current
349        for step in range(int((self.Npoints+1)//2), int(self.Npoints)):
350            self.dac_set(0, cur*self.window[step])
351            delay(self.dt)

364        for step in range(1, int(Npoints)):
365            self.dac_set(0, bottom + (top-bottom)/time*step*dt)
