139    @kernel
140    def measure(self, time, frequency):
141
142        pulse_time = time
143        freq = frequency
144
145        self.core.reset()
146        delay(1*ms)
147        self.core.break_realtime()
148        delay(20*ms)
149        if self.v_sel:
150            self.length = self.vel_sel_time + self.rabi_delay + pulse_time
151        else:
152            self.length = pulse_time
153        self.load_gravity_chirp(self.length + 50*us)
154        self.core.wait_until_mu(now_mu())
155        delay(100*ms)
156
157        if self.v_sel:
158            self.Bragg.set_AOM_freqs([("Bragg2",self.Bragg.freq_Bragg2+self.v_sel_det)])
159        else:
160            self.Bragg.set_AOM_freqs([("Bragg2",freq)])
161        delay(20*ms)
162        self.MOTs.AOMs_off(self.MOTs.AOMs)
163        delay(15*ms)
164
165
166        self.MOTs.rMOT_pulse()
167        delay(self.dipole_load_time)
168
169        self.Bragg.set_AOM_attens([("Dipole",30.0), ("Homodyne",30.0)])
170        self.Bragg.set_AOM_scales([("Dipole",0.6), ("Homodyne",0.1)])
171        self.Bragg.AOMs_off(['Homodyne'])
172
173
174        #exp start
175        self.ttl1.on()
176        if self.v_sel:
177            with parallel:
178                self.Bragg.bragg_pulse(self.vel_sel_time)
179                self.chirp_dds.cpld.io_update.pulse_mu(8)
180            with parallel:
181                delay(self.rabi_delay)
182                self.Bragg.set_AOM_freqs([("Bragg2",freq)])
183            self.Bragg.bragg_pulse(pulse_time)
184
185
186        else:
187            with parallel:
188                self.chirp_dds.cpld.io_update.pulse_mu(8)
189                self.Bragg.bragg_pulse(pulse_time)
190
191        self.ttl1.off()
192
193
194        self.Bragg.set_AOM_attens([("Dipole",self.Bragg.atten_Dipole), ("Homodyne",30.0)])  # keeping on dipole guide
195        self.Bragg.set_AOM_scales([("Dipole",self.Bragg.scale_Dipole)])
196
197        delay(self.drift_time)
198
199        self.MOTs.take_MOT_image(self.Camera)
200        self.Bragg.set_AOM_attens([("Dipole",self.Bragg.atten_Dipole ), ("Homodyne",self.Bragg.atten_Homodyne)])
201        self.Bragg.set_AOM_scales([("Dipole",self.Bragg.scale_Dipole ), ("Homodyne",self.Bragg.scale_Homodyne)])
202        self.Bragg.AOMs_on(['Homodyne'])
203
204
205        delay(50*ms)
206        self.Camera.process_image(bg_sub=True)
207        delay(400*ms)
208        self.MOTs.AOMs_off(['3P0_repump', '3P2_repump', '3D', "Probe"])
209        delay(200*ms)
210        self.chirp_dds.set_cfr1(ram_enable=0)
211        self.chirp_dds.cpld.io_update.pulse_mu(8)
212        delay(10*ms)
213        self.core.wait_until_mu(now_mu())
214        return 0

217    def before_measure(self, point, measurement):
218
219        if point > 1000:
220            time =  self.pulse_time
221        else:
222            time = point
223
224        LENGTH = time
225        if self.v_sel:LENGTH += self.rabi_delay + self.vel_sel_time
226        SAMPLES = 6000
227        SAMPLE_RATE = SAMPLES/LENGTH
228
229        self.wg.sample_rate = SAMPLE_RATE
230
231
232        self.wg.reset_waveform()
233        if self.v_sel:
234            self.wg.add_gaussian_pulse(0, self.vel_sel_time, self.vel_sel_amp, self.vel_sel_time/7)
235            self.wg.add_gaussian_pulse(self.vel_sel_time+self.rabi_delay, time, self.rabi_amp, time/7)
236        else:
237            self.wg.add_gaussian_pulse(0, time, self.rabi_amp, time/7)
238
239
240        t, y = self.wg.get_waveform()
241        val_str = ",".join(map(str, y))
242
243        self.rigol.write(":OUTP1 OFF");
244
245        self.rigol.write(":TRACE:DATA VOLATILE," + val_str)
246        self.rigol.write(f":SOUR1:APPL:USER {1/LENGTH}, {2*self.wg.max}, 0, 0")
247
248        self.rigol.write("SOUR1:BURS ON")
249        self.rigol.write("SOUR1:BURS:MODE TRIG")
250        self.rigol.write(":SOUR1:BURS:NCYC 1 ")
251        self.rigol.write(":SOUR1:BURS:TRIG:SOUR EXT")
252
253
254        self.rigol.write(":OUTP1 ON");
255
256        self.Camera.arm()

272    @kernel
273    def load_gravity_chirp(self, length):
274
275        # loads frequency list
276        self.step_size = int(length/(1024*4*ns))
277        f_range = length*self.doppler_chirp_rate
278        f0 = self.Bragg.freq_Bragg1
279        f_step  = f_range/1023
280        for i in range(1024):
281            self.freq_list[1023-i] = f0+f_step*i
282
283
284        self.chirp_dds.frequency_to_ram(self.freq_list, self.freq_list_ram)
285        self.core.break_realtime()
286
287        self.chirp_dds.set(f0, amplitude=self.Bragg.scale_Bragg1)
288        delay(1 * ms)
289
290        self.chirp_dds.set_cfr1(ram_enable=0)
291        self.chirp_dds.cpld.io_update.pulse_mu(8)
292
293        self.chirp_dds.set_profile_ram(start=0, end=1024-1, step=(self.step_size | (2**6 - 1 ) << 16),
294                                  profile=0, mode=ad9910.RAM_MODE_RAMPUP)
295        self.chirp_dds.cpld.set_profile(0)
296        delay(10*us)
297
298        self.chirp_dds.cpld.io_update.pulse_mu(8)
299        delay(10*us)
300
301        self.chirp_dds.write_ram(self.freq_list_ram)
302        delay(10 * us)
303
304        self.chirp_dds.set_cfr1(internal_profile=0, ram_enable=1, ram_destination=ad9910.RAM_DEST_FTW) # gets the chirp ready to pulse
305        delay(10*ms)
306
307
308        self.core.wait_until_mu(now_mu())
